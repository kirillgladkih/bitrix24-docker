```sh
cp .env.example .env
```

### Заполняем .env директивы
```
PROJECT_PREFIX=<Название проекта>
APP_LOCATION=<Путь до папки с битрикс>
LOGS_LOCATION=<Путь до папки хранения логов>
DATA_LOCATION=<Путь до папки хранения базы данных>
HTTP_PORT=<Порт приложения>
APP_ENV=<test>
MYSQL_HOST=<Хост mysql>
MYSQL_DATABASE=<Название базы mysql>
MYSQL_USER=<Пользователь mysql>
MYSQL_PASSWORD=<Пароль mysql>
```
